import flet as ft import random def main(page: ft.Page): page.title = "Creative Hub Project" page.theme_mode = ft.ThemeMode.DARK page.window_width = 450 page.window_height = 800 page.scroll = "auto" page.padding = 20 # دالة معالجة الأفكار الخمسة def run_logic(feature): text = user_input.value if not text: result_area.value = "⚠️ الرجاء كتابة شيء في الحقل أولاً!" page.update() return if feature == "1": # Persona res = f"👤 تحليل الشخصية لـ ({text}):\nأنت تملك طاقة إبداعية عالية، شخصيتك تشبه المحاربين القدامى بعقلية تكنولوجية حديثة." elif feature == "2": # Dream res = f"🌙 تفسير الحلم:\nرؤيتك لـ ({text}) تدل على طموح كبير يسعى للخروج من حيز التفكير إلى التنفيذ." elif feature == "3": # Future res = f"✉️ رسالة من مستقبلك:\n'يا {text}، أنا فخور بك لأنك لم تستسلم في 2026. نحن الآن نعيش النجاح!'" elif feature == "4": # Aesthetic res = f"🎨 هويتك البصرية:\nبناءً على ({text})، ذوقك يميل للنمط (Cyberpunk) مع لمسات كلاسيكية هادئة." elif feature == "5": # Chaos scenes = ["مدينة طافية فوق السحاب", "مكتبة ضائعة في ثقب أسود", "غابة من الأسلاك المضيئة"] res = f"🎲 مشهد عشوائي:\n{random.choice(scenes)} حيث يوجد {text}." result_area.value = res page.update() # عناصر الواجهة user_input = ft.TextField( label="اكتب فكرتك أو اسمك هنا", border_color="blue", text_align="right", border_radius=15 ) result_area = ft.Text("", color="cyan", size=18, weight="bold", text_align="center") def btn(title, icon, color, key): return ft.ElevatedButton( content=ft.Row([ft.Icon(icon), ft.Text(title)], alignment="center"), on_click=lambda _: run_logic(key), style=ft.ButtonStyle(color=color, shape=ft.RoundedRectangleBorder(radius=10)), height=50 ) # إضافة العناصر للصفحة page.add( ft.Column([ ft.Text("مركز المشاريع الإبداعي", size=28, weight="bold"), ft.Text("ادمج أفكارك الخمسة في مكان واحد", color="grey"), ft.Divider(height=20), user_input, ft.Divider(height=10), btn("توليد الشخصية (Persona)", ft.icons.PERSON, "white", "1"), btn("مفسر الأحلام (Dream)", ft.icons.NIGHTS_STAY, "purple", "2"), btn("رسالة المستقبل (Future)", ft.icons.AUTO_AWESOME, "amber", "3"), btn("الهوية البصرية (Aesthetic)", ft.icons.PALETTE, "green", "4"), btn("المحرك العشوائي (Chaos)", ft.icons.STORM, "red", "5"), ft.Divider(height=30), ft.Container( content=result_area, padding=20, bgcolor=ft.colors.WHITE10, border_radius=15, alignment=ft.alignment.center ) ], horizontal_alignment="center") ) ft.app(target=main)
