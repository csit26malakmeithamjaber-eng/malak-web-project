import turtle import colorsys # إعداد الشاشة screen = turtle.Screen() screen.setup(width=800, height=800) screen.bgcolor('black') # خلفية سوداء لإبراز الألوان screen.title("لوحة فنية برمجية احترافية") # إعدادات الرسم t = turtle.Turtle() t.speed(0) # أقصى سرعة t.width(2) t.hideturtle() # المتغيرات الأساسية للفن hue = 0.0 n = 120 # عدد الأضلاع والتكرار for i in range(300): # تحويل نظام الألوان من HSV إلى RGB للحصول على تدرج احترافي color = colorsys.hsv_to_rgb(hue, 1, 1) t.pencolor(color) hue += 0.005 # سرعة تغير اللون # الرسم الهندسي غير المتوقع t.forward(i * 2) t.right(121) # زاوية الدوران هي السر في شكل الفن t.circle(i, 80) # إضافة انحناءات لتعطي طابعاً فنياً t.left(60) # إنهاء الرسم عند النقر screen.exitonclick()
